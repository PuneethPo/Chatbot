# ingest.py (ingest local docs into FAISS)

import os
import glob
import argparse
from pathlib import Path
from sentence_transformers import SentenceTransformer
import faiss
from PyPDF2 import PdfReader
from tqdm import tqdm
import pickle

EMB_MODEL = "all-mpnet-base-v2"
CHUNK_SIZE = 500
CHUNK_OVERLAP = 50


def clean_text(s: str) -> str:
    return " ".join(s.split())


def pdf_to_text(path: str) -> str:
    reader = PdfReader(path)
    texts = []
    for p in reader.pages:
        try:
            content = p.extract_text()
            if content:
                texts.append(content)
        except Exception:
            continue
    return "\n".join(texts)


def chunk_text(text, size=CHUNK_SIZE, overlap=CHUNK_OVERLAP):
    tokens = text.split()
    i = 0
    chunks = []
    while i < len(tokens):
        chunk = tokens[i:i+size]
        chunks.append(" ".join(chunk))
        i += size - overlap
    return chunks

# Ingest doc and create embeddings 
def ingest_folder(folder, out_dir="faiss_index"):
    model = SentenceTransformer(EMB_MODEL)
    texts = []
    metadatas = []

    for ext in ("*.pdf", "*.txt", "*.md"):
        for file in glob.glob(os.path.join(folder, "**", ext), recursive=True):
            if file.lower().endswith('.pdf'):
                text = pdf_to_text(file)
            else:
                with open(file, 'r', encoding='utf-8', errors='ignore') as f:
                    text = f.read()
            text = clean_text(text)
            if not text.strip():
                continue

            chunks = chunk_text(text)
            for idx, c in enumerate(chunks):
                texts.append(c)
                metadatas.append({"source": file, "chunk": idx})

    print(f"Embedding {len(texts)} chunks using {EMB_MODEL} ...")
    embeddings = model.encode(texts, show_progress_bar=True, convert_to_numpy=True)

    dim = embeddings.shape[1]
    index = faiss.IndexFlatL2(dim)
    index.add(embeddings)

    os.makedirs(out_dir, exist_ok=True)
    faiss.write_index(index, os.path.join(out_dir, "index.faiss"))
    with open(os.path.join(out_dir, "metadata.pkl"), "wb") as f:
        pickle.dump({"texts": texts, "metadatas": metadatas}, f)

    print("✅ Ingest complete. Index & metadata saved to", out_dir)


if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--folder", required=True, help="Path to corpus folder")
    parser.add_argument("--out", default="faiss_index", help="Output index directory")
    args = parser.parse_args()
    ingest_folder(args.folder, out_dir=args.out)
