# Working agent.py and app.py 

import streamlit as st
import re
from vector_store import VectorStore
from agent import SimpleAgent
from tools import should_call_usgs, validate_usgs_streamflow
from rouge_score import rouge_scorer  

# --- Helper: simple sentence split without nltk ---
def truncate_sentences(text: str, max_sentences: int = 4) -> str:
    sentences = re.split(r'(?<=[.!?])\s+', text)
    if len(sentences) > max_sentences:
        return " ".join(sentences[:max_sentences])
    return text

# --- Streamlit page config ---
st.set_page_config(page_title="Ecolab RAG Agent", layout="wide")
st.title("Ecolab — RAG + Tool-calling Agent (demo)")

# --- Initialize agent safely ---
if "agent" not in st.session_state:
    try:
        vs = VectorStore(index_dir="faiss_index")
    except Exception:
        vs = None
    st.session_state.agent = SimpleAgent(vs)
    if vs:
        st.success("Vector store loaded.")
    else:
        st.warning("No FAISS index found. Agent will answer from API only.")

# --- Query input ---
query = st.text_input("Ask a question about water / sites / treatment / measurements")

if query and st.button("Ask"):
    with st.spinner("Thinking..."):
        # Get chatbot answer and RAG context
        chatbot_answer, context = st.session_state.agent.answer(query, return_context=True)

        # Truncate Chatbot Answer
        chatbot_answer = truncate_sentences(chatbot_answer, max_sentences=4)

        # Determine if question requires USGS API validation
        requires_usgs_validation = should_call_usgs(query)
        validated_answer = None

        if requires_usgs_validation:
            match = re.search(r"\b(\d{8})\b", query)
            if match:
                site_id = match.group(1)
                actual_value = validate_usgs_streamflow(site_id)
                if isinstance(actual_value, float):
                    validated_answer = (
                        f"The streamflow at site {site_id} is {actual_value} cubic feet per second (ft³/s), "
                        f"validated via USGS API."
                    )
                else:
                    validated_answer = f"USGS validation error: {actual_value}"

            if validated_answer:
                validated_answer = truncate_sentences(validated_answer, max_sentences=4)

        # Truncate RAG context
        if context.strip():
            context = truncate_sentences(context, max_sentences=4)

    # --- Display results ---
    st.markdown("### 🤖 Chatbot Answer")
    st.markdown(
        f"<div style='width:90%; line-height:1.5; font-size:16px;'>{chatbot_answer}</div>",
        unsafe_allow_html=True
    )

    if requires_usgs_validation:
        st.markdown("### ✅ Validated Answer")
        st.markdown(
            f"<div style='width:90%; line-height:1.5; font-size:16px;'>{validated_answer}</div>",
            unsafe_allow_html=True
        )
    else:
        st.markdown("### 📚 Retrieved Context (RAG)")
        if context.strip():
            st.markdown(
                f"<div style='width:90%; line-height:1.5; font-size:16px;'>{context}</div>",
                unsafe_allow_html=True
            )
        else:
            st.info("No context retrieved (index empty or not relevant).")

    # --- 🔹 ROUGE Score Implementation ---
    if context.strip():
        scorer = rouge_scorer.RougeScorer(["rouge1", "rougeL"], use_stemmer=True)
        scores = scorer.score(context, chatbot_answer)

        st.markdown("### 📊 ROUGE Evaluation")
        st.write({
            "ROUGE-1 (F1)": round(scores["rouge1"].fmeasure, 3),
            "ROUGE-L (F1)": round(scores["rougeL"].fmeasure, 3)
        })

st.markdown("---")
st.markdown(
    "**Instructions:** 1) Put PDFs/TXT/MD into `./corpus/` "
    "2) Run `python ingest.py --folder corpus` "
    "3) Run `streamlit run app.py`"
)

# agent.py - SimpleAgent without NLTK
import re
from dotenv import load_dotenv
import os
from openai import AzureOpenAI
import json
from vector_store import VectorStore
from tools import call_usgs_iv, should_call_usgs

# Load environment variables
load_dotenv()

# Azure OpenAI credentials
AZURE_OPENAI_ENDPOINT = os.getenv("ENDPOINT_URL")
AZURE_OPENAI_API_KEY = os.getenv("AZURE_OPENAI_API_KEY")
AZURE_OPENAI_DEPLOYMENT = os.getenv("DEPLOYMENT_NAME")
AZURE_OPENAI_API_VERSION = os.getenv("API_VERSION")

# Initialize Azure client
client = AzureOpenAI(
    api_key=AZURE_OPENAI_API_KEY,
    api_version=AZURE_OPENAI_API_VERSION,
    azure_endpoint=AZURE_OPENAI_ENDPOINT,
)

# --- Helper: simple sentence split without nltk ---
def truncate_sentences(text: str, max_sentences: int = 4) -> str:
    sentences = re.split(r'(?<=[.!?])\s+', text)
    if len(sentences) > max_sentences:
        return " ".join(sentences[:max_sentences])
    return text

class SimpleAgent:
    def __init__(self, vs: VectorStore = None):
        self.vs = vs

    def answer(self, question: str, return_context: bool = False):
        # Retrieve top passages
        if self.vs:
            retrieved = self.vs.query(question, k=4)
            passages = "\n---\n".join([r['text'] for r in retrieved]) if retrieved else ""
        else:
            retrieved, passages = [], ""

        # Check if question needs USGS API
        api_response = None
        if should_call_usgs(question):
            m = re.search(r"\b(\d{3,})\b", question)
            if m:
                site = m.group(1)
                try:
                    api_response = call_usgs_iv(site)
                except Exception as e:
                    api_response = {"error": str(e)}
            else:
                api_response = {"hint": "No site id detected. Provide numeric site id (e.g., 01234567)."}

        # Prepare prompt for Azure OpenAI
        system_msg = "You are an expert water resources assistant."
        user_msg = (
            f"Answer the question concisely in 3-4 sentences. "
            f"Use complete sentences. Be clear and paragraph-style.\n\n"
            f"Question: {question}\n\n"
            f"Passages:\n{passages}\n\n"
            f"API:\n{json.dumps(api_response) if api_response else 'None'}"
        )

        # Azure OpenAI chat completion
        try:
            completion = client.chat.completions.create(
                model=AZURE_OPENAI_DEPLOYMENT,
                messages=[
                    {"role": "system", "content": system_msg},
                    {"role": "user", "content": user_msg},
                ],
                max_tokens=300,
                temperature=0.0,
            )
            answer = completion.choices[0].message.content.strip()
            answer = truncate_sentences(answer, max_sentences=4)
        except Exception as e:
            answer = f"Error calling Azure OpenAI: {e}"

        # Truncate RAG context
        if passages:
            passages = truncate_sentences(passages, max_sentences=4)

        if return_context:
            return answer, passages
        return answer
